       // ============================================================
       // =============== PACKAGE SELECT FUNCTION ====================

       function selectPackage(packageId) {
           const data = packageData[packageId];
           currentPackage = packageId;
           currentMediaIndex = 0;

           // Update radio button styling
           document.querySelectorAll('.package-option').forEach(opt => {
               opt.classList.remove('selected');
           });
           document.getElementById('option-' + packageId).classList.add('selected');
           document.getElementById('option-' + packageId).querySelector('input').checked = true;

           // Update hidden input
           document.getElementById('selectedPackageInput').value = data.name;

           // Update preview content with animation
           const preview = document.getElementById('packagePreview');
           preview.style.opacity = '0.7';
           preview.style.transform = 'scale(0.98)';

           setTimeout(() => {
               // Update text content
               document.getElementById('previewBadge').textContent = data.badge;
               document.getElementById('previewTitle').textContent = data.name;
               document.getElementById('previewPrice').innerHTML = data.price + '<span>/person</span>';

               // Update features
               const featuresList = document.getElementById('previewFeatures');
               featuresList.innerHTML = data.features.map(f => `<li>${f}</li>`).join('');

               // Update media (Photo/Video)
               updateMainMedia();

               // Update thumbnails
               updateThumbnails();

               // Restore opacity
               preview.style.opacity = '1';
               preview.style.transform = 'scale(1)';
           }, 200);
       }

       // ============================================================
       // =============== UPDATE MAIN MEDIA (PHOTO/VIDEO) ============
       // ============================================================

       function updateMainMedia() {
           const data = packageData[currentPackage];
           const media = data.mediaGallery[currentMediaIndex];
           const videoEl = document.getElementById('previewVideo');
           const imgEl = document.getElementById('previewImage');
           const indicator = document.getElementById('mediaIndicator');

           if (media.type === 'video' && media.src) {
               // Show video
               videoEl.style.display = 'block';
               imgEl.style.display = 'none';
               videoEl.src = media.src;
               videoEl.play();
               indicator.innerHTML = '<i class="fas fa-video"></i> VIDEO';
           } else {
               // Show image
               videoEl.style.display = 'none';
               imgEl.style.display = 'block';
               imgEl.src = media.src;
               videoEl.pause();
               indicator.innerHTML = '<i class="fas fa-image"></i> PHOTO';
           }
       }

       // ============================================================
       // =============== UPDATE THUMBNAILS ==========================
       // ============================================================

       function updateThumbnails() {
           const data = packageData[currentPackage];
           const container = document.getElementById('previewThumbnails');
           
           container.innerHTML = data.thumbnails.map((thumb, index) => `
               <div class="preview-thumb ${index === 0 ? 'active' : ''}" onclick="changeMedia(${index})">
                   <img src="${thumb}" alt="View ${index + 1}">
               </div>
           `).join('');
       }

       // ============================================================
       // =============== CHANGE MEDIA FROM THUMBNAIL ================
       // ============================================================

       function changeMedia(index) {
           currentMediaIndex = index;
           
           // Update thumbnail active state
           document.querySelectorAll('.preview-thumb').forEach((thumb, i) => {
               thumb.classList.toggle('active', i === index);
           });

           // Update main media with fade effect
           const mediaContainer = document.getElementById('previewMedia');
           mediaContainer.style.opacity = '0.5';
           
           setTimeout(() => {
               updateMainMedia();
               mediaContainer.style.opacity = '1';
           }, 200);
       }
